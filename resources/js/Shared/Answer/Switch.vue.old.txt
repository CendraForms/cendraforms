<template>
  <label v-text="left" for="left" />
  <input
    id="left"
    type="checkbox"
    v-model="defaultVal"
    @click="toggleChecked(left)"
  >

  <label v-text="right" for="right" />
  <input
    id="right"
    type="checkbox"
    v-model="right.selected"
    @click="toggleChecked(right)"
  >
</template>

<script setup>
import { ref } from 'vue'

const props = defineProps({
  left: String,
  right: String,
  defaultVal: String,
})

const left = ref(props.left)
const right = ref(props.right)
const defaultVal = ref(props.defaultVal)

const toggleChecked = (clickedBtn) => {
  defaultVal.value = clickedBtn;

  left.value.selected = false;
  right.value.selected = false;

  clickedBtn.selected = true;
}
</script>
